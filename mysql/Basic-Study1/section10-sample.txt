■符号による論理式
SELECT * FROM order_details WHERE product_id = 3;

SELECT * FROM order_details WHERE product_id != 3;

SELECT * FROM order_details WHERE product_id <> 3;
 
■不等号による論理式
SELECT * FROM products;

SELECT * FROM products WHERE price > 40;

SELECT * FROM products WHERE price >= 40;

SELECT * FROM products WHERE price < 40;

SELECT * FROM products WHERE price <= 40;

SELECT * FROM products WHERE price <> 40;
